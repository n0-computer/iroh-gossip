26/// A message identifier, which is the message content's blake3 hash.
27#[derive(Serialize, Deserialize, Clone, Hash, Copy, PartialEq, Eq, MaxSize)]
28pub struct MessageId([u8; 32]);
29idbytes_impls!(MessageId, "MessageId");
30
31impl MessageId {
32    /// Create a `[MessageId]` by hashing the message content.
33    ///
34    /// This hashes the input with [`blake3::hash`].
35    pub fn from_content(message: &[u8]) -> Self {
36        Self::from(blake3::hash(message))
37    }
38}

40/// Events Plumtree is informed of from the peer sampling service and IO layer.
41#[derive(Debug)]
42pub enum InEvent<PI> {
43    /// A [`Message`] was received from the peer.
44    RecvMessage(PI, Message),
45    /// Broadcast the contained payload to the given scope.
46    Broadcast(Bytes, Scope),
47    /// A timer has expired.
48    TimerExpired(Timer),
49    /// New member `PI` has joined the topic.
50    NeighborUp(PI),
51    /// Peer `PI` has disconnected from the topic.
52    NeighborDown(PI),
53}

55/// Events Plumtree emits.
56#[derive(Debug, PartialEq, Eq)]
57pub enum OutEvent<PI> {
58    /// Ask the IO layer to send a [`Message`] to peer `PI`.
59    SendMessage(PI, Message),
60    /// Schedule a [`Timer`].
61    ScheduleTimer(Duration, Timer),
62    /// Emit an [`Event`] to the application.
63    EmitEvent(Event<PI>),
64}
65
66/// Kinds of timers Plumtree needs to schedule.
67#[derive(Clone, Debug, Eq, PartialEq)]
68pub enum Timer {
69    /// Request the content for [`MessageId`] by sending [`Message::Graft`].
70    ///
71    /// The message will be sent to a peer that sent us an [`Message::IHave`] for this [`MessageId`],
72    /// which will send us the message content in reply and also move the peer into the eager set.
73    /// Will be a no-op if the message for [`MessageId`] was already received from another peer by now.
74    SendGraft(MessageId),
75    /// Dispatch the [`Message::IHave`] in our lazy push queue.
76    DispatchLazyPush,
77    /// Evict the message cache
78    EvictCache,
79}

135/// Messages that we can send and receive from peers within the topic.
136#[derive(Debug, Serialize, Deserialize, Clone, PartialEq, Eq)]
137pub enum Message {
138    /// When receiving Gossip, emit as event and forward full message to eager peer and (after a
139    /// delay) message IDs to lazy peers.
140    Gossip(Gossip),
141    /// When receiving Prune, move the peer from the eager to the lazy set.
142    Prune,
143    /// When receiving Graft, move the peer to the eager set and send the full content for the
144    /// included message ID.
145    Graft(Graft),
146    /// When receiving IHave, do nothing initially, and request the messages for the included
147    /// message IDs after some time if they aren't pushed eagerly to us.
148    IHave(Vec<IHave>),
149}

151/// Payload messages transmitted by the protocol.
152#[derive(Serialize, Deserialize, Clone, derive_more::Debug, PartialEq, Eq)]
153pub struct Gossip {
154    /// Id of the message.
155    id: MessageId,
156    /// Message contents.
157    #[debug("<{}b>", content.len())]
158    content: Bytes,
159    /// Scope to broadcast to.
160    scope: DeliveryScope,
161}

163impl Gossip {
164    fn round(&self) -> Option<Round> {
165        match self.scope {
166            DeliveryScope::Swarm(round) => Some(round),
167            DeliveryScope::Neighbors => None,
168        }
169    }
170}

172/// The scope to deliver the message to.
173#[derive(Serialize, Deserialize, Clone, Debug, PartialEq, Eq, Ord, PartialOrd, Copy)]
174pub enum DeliveryScope {
175    /// This message was received from the swarm, with a distance (in hops) travelled from the
176    /// original broadcaster.
177    Swarm(Round),
178    /// This message was received from a direct neighbor that broadcasted the message to neighbors
179    /// only.
180    Neighbors,
181}
182
183impl DeliveryScope {
184    /// Whether this message was directly received from its publisher.
185    pub fn is_direct(&self) -> bool {
186        matches!(self, Self::Neighbors | Self::Swarm(Round(0)))
187    }
188}
189
190/// The broadcast scope of a gossip message.
191#[derive(Serialize, Deserialize, Clone, Debug, PartialEq, Eq, Ord, PartialOrd, Copy)]
192pub enum Scope {
193    /// The message is broadcast to all peers in the swarm.
194    Swarm,
195    /// The message is broadcast only to the immediate neighbors of a peer.
196    Neighbors,
197}
198
199impl Gossip {
200    /// Get a clone of this `Gossip` message and increase the delivery round by 1.
201    pub fn next_round(&self) -> Option<Gossip> {
202        match self.scope {
203            DeliveryScope::Neighbors => None,
204            DeliveryScope::Swarm(round) => Some(Gossip {
205                id: self.id,
206                content: self.content.clone(),
207                scope: DeliveryScope::Swarm(round.next()),
208            }),
209        }
210    }
211
212    /// Validate that the message id is the blake3 hash of the message content.
213    pub fn validate(&self) -> bool {
214        let expected = MessageId::from_content(&self.content);
215        expected == self.id
216    }
217}

239/// Controls when to prune peers from eager to lazy push sets.
240///
241/// The plumtree protocol optimizes bandwidth by moving peers that send duplicate messages
242/// from eager (full payload) to lazy (message IDs only). In small swarms or burst
243/// scenarios, this can hurt performance.
244#[derive(Clone, Debug, Serialize, Deserialize, PartialEq, Eq)]
245pub enum PruneMode {
246    /// Prune immediately on first duplicate message (default).
247    /// Best for large swarms with stable topology.
248    Immediate,
249    /// Prune after receiving N duplicate messages from same peer.
250    /// Best for burst scenarios or small swarms. Recommended: 3-10.
251    AfterDuplicates(usize),
252    /// Never prune, keep all peers in eager set.
253    /// Use for very small swarms (2-3 nodes) or when reliability > bandwidth.
254    Disabled,
255}
256
257impl Default for PruneMode {
258    fn default() -> Self {
259        // Default to pruning after 3 duplicates for better handling of burst scenarios.
260        Self::AfterDuplicates(3)
261    }
262}

264/// Configuration for the gossip broadcast layer.
265///
266/// Currently, the expectation is that the configuration is the same for all peers in the
267/// network (as recommended in the paper).
268#[derive(Clone, Debug, Serialize, Deserialize)]
269#[serde(default)]
270pub struct Config {
271    /// When receiving an `IHave` message, this timeout is registered. If the message for the
272    /// `IHave` was not received once the timeout is expired, a `Graft` message is sent to the
273    /// peer that sent us the `IHave` to request the message payload.
274    ///
275    /// The plumtree paper notes:
276    /// > The timeout value is a protocol parameter that should be configured considering the
277    /// > diameter of the overlay and a target maximum recovery latency, defined by the application
278    /// > requirements. (p.8)
279    pub graft_timeout_1: Duration,
280    /// This timeout is registered when sending a `Graft` message. If a reply has not been
281    /// received once the timeout expires, we send another `Graft` message to the next peer that
282    /// sent us an `IHave` for this message.
283    ///
284    /// The plumtree paper notes:
285    /// > This second timeout value should be smaller that the first, in the order of an average
286    /// > round trip time to a neighbor.
287    pub graft_timeout_2: Duration,
288    /// Timeout after which `IHave` messages are pushed to peers.
289    pub dispatch_timeout: Duration,
290    /// The protocol performs a tree optimization, which promotes lazy peers to eager peers if the
291    /// `Message::IHave` messages received from them have a lower number of hops from the
292    /// message's origin as the `InEvent::Broadcast` messages received from our eager peers. This
293    /// parameter is the number of hops that the lazy peers must be closer to the origin than our
294    /// eager peers to be promoted to become an eager peer.
295    pub optimization_threshold: Round,
296
297    /// Duration for which to keep gossip messages in the internal message cache.
298    ///
299    /// Messages broadcast from this node or received from other nodes are kept in an internal
300    /// cache for this duration before being evicted. If this is too low, other nodes will not be
301    /// able to retrieve messages once they need them. If this is high, the cache will grow.
302    ///
303    /// Should be at least around several round trip times to peers.
304    pub message_cache_retention: Duration,
305
306    /// Duration for which to keep the `MessageId`s for received messages.
307    ///
308    /// Should be at least as long as [`Self::message_cache_retention`], usually will be longer to
309    /// not accidentally receive messages multiple times.
310    pub message_id_retention: Duration,
311
312    /// How often the internal caches will be checked for expired items.
313    pub cache_evict_interval: Duration,
314
315    /// Controls when peers are pruned from eager to lazy push sets.
316    ///
317    /// - `Immediate`: Prune on first duplicate
318    /// - `AfterDuplicates(n)`: More tolerant, prune after n duplicates
319    /// - `Disabled`: Never prune, useful for 2-3 node swarms
320    pub prune_mode: PruneMode,
321
322    /// Cooldown period after promoting a peer to eager set.
323    ///
324    /// During this period, the peer will not be pruned on duplicate receipt.
325    /// Prevents aggressive pruning during rapid message bursts.
326    pub prune_cooldown: Duration,
327
328    /// Minimum number of eager peers to maintain.
329    ///
330    /// Pruning will be skipped if it would reduce eager peers below this threshold.
331    /// Provides a safety floor to prevent mesh fragmentation.
332    pub min_eager_peers: usize,
333
334    /// Retention period for disconnected lazy peers.
335    ///
336    /// When a neighbor disconnects (HyParView shuffle), they stay in lazy_push_peers
337    /// for this duration. If they don't reconnect within this time, they're removed.
338    /// Set to 2x HyParView shuffle_interval (60s) by default.
339    pub lazy_peer_retention: Duration,
340}
341
342impl Default for Config {
343    /// Sensible defaults for the plumtree configuration
344    //
345    // TODO: Find out what good defaults are for the three timeouts here. Current numbers are
346    // guesses that need validation. The paper does not have concrete recommendations for these
347    // numbers.
348    fn default() -> Self {
349        Self {
350            // Graft timeout: when we receive IHave, wait for this duration before sending Graft.
351            // Higher value provides more tolerance for unstable connections under load. Lower
352            // value provides faster recovery of missing messages.
353            graft_timeout_1: Duration::from_millis(80),
354
355            // Second graft timeout for retry.
356            graft_timeout_2: Duration::from_millis(40),
357
358            // Again, paper does not tell a recommended number here. Likely should be quite small,
359            // as to not delay messages without need. This would also be the time frame in which
360            // `IHave`s are aggregated to save on packets.
361            //
362            // Eartstar dispatches immediately from my reading.
363            dispatch_timeout: Duration::from_millis(5),
364
365            // This number comes from experiment settings the plumtree paper (p. 12)
366            optimization_threshold: Round(7),
367
368            // This is a certainly-high-enough value for usual operation.
369            message_cache_retention: Duration::from_secs(30),
370            message_id_retention: Duration::from_secs(90),
371            cache_evict_interval: Duration::from_secs(1),
372
373            // Default to pruning after 3 duplicates for better handling of burst scenarios.
374            prune_mode: PruneMode::default(),
375            // 1s cooldown prevents cascade pruning under load
376            prune_cooldown: Duration::from_secs(1),
377            // Keep at least 2 eager peers to prevent mesh fragmentation
378            min_eager_peers: 2,
379            // 2x HyParView shuffle_interval (30s) for stale lazy peer cleanup
380            lazy_peer_retention: Duration::from_secs(60),
381        }
382    }
383}

400/// State of the plumtree.
401#[derive(Debug)]
402pub struct State<PI> {
403    /// Our address.
404    me: PI,
405    /// Configuration for this plumtree.
406    config: Config,
407
408    /// Set of peers used for payload exchange.
409    pub(crate) eager_push_peers: BTreeSet<PI>,
410    /// Set of peers used for control message exchange.
411    pub(crate) lazy_push_peers: BTreeSet<PI>,
412
413    lazy_push_queue: BTreeMap<PI, Vec<IHave>>,
414
415    /// Messages for which a [`MessageId`] has been seen via a [`Message::IHave`] but we have not
416    /// yet received the full payload. For each, we store the peers that have claimed to have this
417    /// message.
418    missing_messages: HashMap<MessageId, VecDeque<(PI, Round)>>,
419    /// Messages for which the full payload has been seen.
420    received_messages: TimeBoundCache<MessageId, ()>,
421    /// Payloads of received messages.
422    cache: TimeBoundCache<MessageId, Gossip>,
423    /// Messages that we originated via broadcast() - don't prune peers for relaying these back.
424    originated_messages: TimeBoundCache<MessageId, ()>,
425
426    /// Message ids for which a [`Timer::SendGraft`] has been scheduled.
427    graft_timer_scheduled: HashSet<MessageId>,
428    /// Whether a [`Timer::DispatchLazyPush`] has been scheduled.
429    dispatch_timer_scheduled: bool,
430
431    /// Set to false after the first message is received. Used for initial timer scheduling.
432    init: bool,
433
434    /// [`Stats`] of this plumtree.
435    pub(crate) stats: Stats,
436
437    max_message_size: usize,
438
439    /// Tracks duplicate message counts per peer for AfterDuplicates prune mode.
440    duplicate_counts: HashMap<PI, usize>,
441    /// Track when peers were last promoted to eager set for prune cooldown.
442    eager_promotion_times: HashMap<PI, Instant>,
443    /// Track when peers were disconnected (removed from eager, kept in lazy).
444    /// Used to cleanup stale lazy peers after lazy_peer_retention.

448impl<PI: PeerIdentity> State<PI> {
449    /// Initialize the [`State`] of a plumtree.
450    pub fn new(me: PI, config: Config, max_message_size: usize) -> Self {
451        Self {
452            me,
453            eager_push_peers: Default::default(),
454            lazy_push_peers: Default::default(),
455            lazy_push_queue: Default::default(),
456            config,
457            missing_messages: Default::default(),
458            received_messages: Default::default(),
459            graft_timer_scheduled: Default::default(),
460            dispatch_timer_scheduled: false,
461            cache: Default::default(),
462            originated_messages: Default::default(),
463            init: false,
464            stats: Default::default(),
465            max_message_size,
466            duplicate_counts: Default::default(),
467            eager_promotion_times: Default::default(),
468            disconnected_lazy_times: Default::default(),
469        }
470    }
471
472    /// Handle an [`InEvent`].
473    pub fn handle(&mut self, event: InEvent<PI>, now: Instant, io: &mut impl IO<PI>) {
474        if !self.init {
475            self.init = true;
476            self.on_evict_cache_timer(now, io)
477        }
478        match event {
479            InEvent::RecvMessage(from, message) => self.handle_message(from, message, now, io),
480            InEvent::Broadcast(data, scope) => self.broadcast(data, scope, now, io),
481            InEvent::NeighborUp(peer) => self.on_neighbor_up(peer, now),
482            InEvent::NeighborDown(peer) => self.on_neighbor_down(peer, now),
483            InEvent::TimerExpired(timer) => match timer {
484                Timer::DispatchLazyPush => self.on_dispatch_timer(io),
485                Timer::SendGraft(id) => {
486                    self.on_send_graft_timer(id, now, io);
487                }
488                Timer::EvictCache => self.on_evict_cache_timer(now, io),
489            },
490        }
491    }
492
493    /// Get access to the [`Stats`] of the plumtree.
494    pub fn stats(&self) -> &Stats {
495        &self.stats
496    }
497
498    /// Handle receiving a [`Message`].
499    fn handle_message(&mut self, sender: PI, message: Message, now: Instant, io: &mut impl IO<PI>) {
500        if matches!(message, Message::Gossip(_)) {
501            self.stats.payload_messages_received += 1;
502        } else {
503            self.stats.control_messages_received += 1;
504        }
505        match message {
506            Message::Gossip(details) => self.on_gossip(sender, details, now, io),
507            Message::Prune => self.on_prune(sender),
508            Message::IHave(details) => self.on_ihave(sender, details, io),
509            Message::Graft(details) => self.on_graft(sender, details, now, io),
510        }
511    }
512
513    /// Dispatches messages from lazy queue over to lazy peers.
514    fn on_dispatch_timer(&mut self, io: &mut impl IO<PI>) {
515        let chunk_size = self.max_message_size
516            // Space for discriminator
517            - 1
518            // Space for length prefix
519            - 2;
520        let chunk_len = chunk_size / IHave::POSTCARD_MAX_SIZE;
521        while let Some((peer, list)) = self.lazy_push_queue.pop_first() {
522            for chunk in list.chunks(chunk_len) {
523                io.push(OutEvent::SendMessage(peer, Message::IHave(chunk.to_vec())));
524            }
525        }
526
527        self.dispatch_timer_scheduled = false;
528    }
529
530    /// Send a gossip message.
531    ///
532    /// Will be pushed in full to eager peers.
533    /// Pushing the message id to the lazy peers is delayed by a timer.
534    fn broadcast(&mut self, content: Bytes, scope: Scope, now: Instant, io: &mut impl IO<PI>) {
535        let id = MessageId::from_content(&content);
536        let scope = match scope {
537            Scope::Neighbors => DeliveryScope::Neighbors,
538            Scope::Swarm => DeliveryScope::Swarm(Round(0)),
539        };
540        let message = Gossip { id, content, scope };
541        let me = self.me;
542        if let DeliveryScope::Swarm(_) = scope {
543            self.received_messages
544                .insert(id, (), now + self.config.message_id_retention);
545            // Track that we originated this message - don't prune peers for relaying it back
546            self.originated_messages
547                .insert(id, (), now + self.config.message_id_retention);
548            self.cache.insert(
549                id,
550                message.clone(),
551                now + self.config.message_cache_retention,
552            );
553            self.lazy_push(message.clone(), &me, io);
554        }
555
556        self.eager_push(message.clone(), &me, io);
557    }
558
559    /// Handle receiving a [`Message::Gossip`].
560    fn on_gossip(&mut self, sender: PI, message: Gossip, now: Instant, io: &mut impl IO<PI>) {
561        // Validate that the message id is the blake3 hash of the message content.
562        if !message.validate() {
563            // TODO: Do we want to take any measures against the sender if we received a message
564            // with a spoofed message id?
565            warn!(
566                peer = ?sender,
567                "Received a message with spoofed message id ({})", message.id
568            );
569            return;
570        }
571
572        // if we already received this message: consider pruning based on config
573        if self.received_messages.contains_key(&message.id) {
574            debug!("duplicate from {:?}, msg={}", sender, message.id);
575            self.maybe_prune_on_duplicate(sender, &message.id, now, io);
576        // otherwise store the message, emit to application and forward to peers
577        } else {
578            // Reset duplicate count on successful message delivery
579            self.duplicate_counts.remove(&sender);
580            // Refresh cooldown for active eager peers - prevents timing-sensitive pruning
581            if self.eager_push_peers.contains(&sender) {
582                self.eager_promotion_times.insert(sender, now);
583            }
584            if let DeliveryScope::Swarm(prev_round) = message.scope {
585                // insert the message in the list of received messages
586                self.received_messages.insert(
587                    message.id,
588                    (),
589                    now + self.config.message_id_retention,
590                );
591                // increase the round for forwarding the message, and add to cache
592                // to reply to Graft messages later
593                // TODO: add callback/event to application to get missing messages that were received before?
594                let message = message.next_round().expect("just checked");
595
596                self.cache.insert(
597                    message.id,
598                    message.clone(),
599                    now + self.config.message_cache_retention,
600                );
601                // push the message to our peers
602                self.eager_push(message.clone(), &sender, io);
603                self.lazy_push(message.clone(), &sender, io);
604                // cleanup places where we track missing messages
605                self.graft_timer_scheduled.remove(&message.id);
606                let previous_ihaves = self.missing_messages.remove(&message.id);
607                // do the optimization step from the paper
608                if let Some(previous_ihaves) = previous_ihaves {
609                    self.optimize_tree(&sender, &message, previous_ihaves, now, io);
610                }
611                self.stats.max_last_delivery_hop =
612                    self.stats.max_last_delivery_hop.max(prev_round.0);
613            }
614
615            // emit event to application
616            io.push(OutEvent::EmitEvent(Event::Received(
617                GossipEvent::from_message(&message, sender),
618            )));
619        }
620    }
621
622    /// Optimize the tree by pruning the `sender` of a [`Message::Gossip`] if we previously
623    /// received a [`Message::IHave`] for the same message with a much lower number of delivery
624    /// hops from the original broadcaster of the message.
625    ///
626    /// See [Config::optimization_threshold].
627    fn optimize_tree(
628        &mut self,
629        gossip_sender: &PI,
630        message: &Gossip,
631        previous_ihaves: VecDeque<(PI, Round)>,
632        now: Instant,
633        io: &mut impl IO<PI>,
634    ) {
635        let round = message.round().expect("only called for swarm messages");
636        let best_ihave = previous_ihaves
637            .iter()
638            .min_by(|(_a_peer, a_round), (_b_peer, b_round)| a_round.cmp(b_round))
639            .copied();
640
641        if let Some((ihave_peer, ihave_round)) = best_ihave {
642            if (ihave_round < round) && (round - ihave_round) >= self.config.optimization_threshold
643            {
644                // Graft the sender of the IHave, but only if it's not already eager.
645                if !self.eager_push_peers.contains(&ihave_peer) {
646                    let message = Message::Graft(Graft {
647                        id: None,
648                        round: ihave_round,
649                    });
650                    self.add_eager_at(ihave_peer, now);
651                    io.push(OutEvent::SendMessage(ihave_peer, message));
652                }
653                // Prune the sender of the Gossip.
654                // Only prune immediately in Immediate mode; AfterDuplicates uses duplicate path.
655                if matches!(self.config.prune_mode, PruneMode::Immediate)
656                    && !self.is_in_prune_cooldown(gossip_sender, now)
657                    && self.eager_push_peers.len() > self.config.min_eager_peers
658                {
659                    self.add_lazy(*gossip_sender);
660                    io.push(OutEvent::SendMessage(*gossip_sender, Message::Prune));
661                }
662            }
663        }
664    }
665
666    /// Handle receiving a [`Message::Prune`].
667    fn on_prune(&mut self, sender: PI) {
668        self.add_lazy(sender);
669    }
670
671    /// Handle receiving a [`Message::IHave`].
672    ///
673    /// > When a node receives a IHAVE message, it simply marks the corresponding message as
674    /// > missing It then starts a timer, with a predefined timeout value, and waits for the missing
675    /// > message to be received via eager push before the timer expires. The timeout value is a
676    /// > protocol parameter that should be configured considering the diameter of the overlay and a
677    /// > target maximum recovery latency, defined by the application requirements. This is a
678    /// > parameter that should be statically configured at deployment time. (p8)
679    fn on_ihave(&mut self, sender: PI, ihaves: Vec<IHave>, io: &mut impl IO<PI>) {
680        for ihave in ihaves {
681            if !self.received_messages.contains_key(&ihave.id) {
682                self.missing_messages
683                    .entry(ihave.id)
684                    .or_default()
685                    .push_back((sender, ihave.round));
686
687                if !self.graft_timer_scheduled.contains(&ihave.id) {
688                    self.graft_timer_scheduled.insert(ihave.id);
689                    io.push(OutEvent::ScheduleTimer(
690                        self.config.graft_timeout_1,
691                        Timer::SendGraft(ihave.id),
692                    ));
693                }
694            }
695        }
696    }
697
698    /// A scheduled [`Timer::SendGraft`] has reached it's deadline.
699    fn on_send_graft_timer(&mut self, id: MessageId, now: Instant, io: &mut impl IO<PI>) {
700        self.graft_timer_scheduled.remove(&id);
701        // if the message was received before the timer ran out, there is no need to request it
702        // again
703        if self.received_messages.contains_key(&id) {
704            return;
705        }
706        // get the first peer that advertised this message
707        let entry = self
708            .missing_messages
709            .get_mut(&id)
710            .and_then(|entries| entries.pop_front());
711        if let Some((peer, round)) = entry {
712            self.add_eager_at(peer, now);
713            let message = Message::Graft(Graft {
714                id: Some(id),
715                round,
716            });
717            io.push(OutEvent::SendMessage(peer, message));
718
719            // "when a GRAFT message is sent, another timer is started to expire after a certain timeout,
720            // to ensure that the message will be requested to another neighbor if it is not received
721            // meanwhile. This second timeout value should be smaller that the first, in the order of
722            // an average round trip time to a neighbor." (p9)
723            io.push(OutEvent::ScheduleTimer(
724                self.config.graft_timeout_2,
725                Timer::SendGraft(id),
726            ));
727        }
728    }
729
730    /// Handle receiving a [`Message::Graft`].
731    fn on_graft(&mut self, sender: PI, details: Graft, now: Instant, io: &mut impl IO<PI>) {
732        self.add_eager_at(sender, now);
733        if let Some(id) = details.id {
734            if let Some(message) = self.cache.get(&id) {
735                io.push(OutEvent::SendMessage(
736                    sender,
737                    Message::Gossip(message.clone()),
738                ));
739            } else {
740                debug!(?id, peer=?sender, "on_graft failed to graft: message not in cache");
741            }
742        }
743    }
744
745    /// Handle a [`InEvent::NeighborUp`] when a peer joins the topic.
746    fn on_neighbor_up(&mut self, peer: PI, now: Instant) {
747        self.add_eager_at(peer, now);
748    }
749
750    /// Handle a [`InEvent::NeighborDown`] when a peer leaves the topic.
751    /// > When a neighbor is detected to leave the overlay, it is simple removed from the
752    /// > membership. Furthermore, the record of IHAVE messages sent from failed members is deleted
753    /// > from the missing history. (p9)
754    fn on_neighbor_down(&mut self, peer: PI, now: Instant) {
755        self.missing_messages.retain(|_message_id, ihaves| {
756            ihaves.retain(|(ihave_peer, _round)| *ihave_peer != peer);
757            !ihaves.is_empty()
758        });
759        self.eager_push_peers.remove(&peer);
760        // Keep peer in lazy_push_peers for recovery after HyParView shuffle.
761        // Track disconnect time for stale peer cleanup.
762        if self.lazy_push_peers.contains(&peer) {
763            self.disconnected_lazy_times.insert(peer, now);
764        }
765        self.duplicate_counts.remove(&peer);
766        self.eager_promotion_times.remove(&peer);
767    }
768
769    fn on_evict_cache_timer(&mut self, now: Instant, io: &mut impl IO<PI>) {
770        self.cache.expire_until(now);
771        self.originated_messages.expire_until(now);
772        self.expire_promotion_times(now);
773        self.expire_disconnected_lazy_peers(now);
774        io.push(OutEvent::ScheduleTimer(
775            self.config.cache_evict_interval,
776            Timer::EvictCache,
777        ));
778    }
779
780    fn expire_disconnected_lazy_peers(&mut self, now: Instant) {
781        let retention = self.config.lazy_peer_retention;
782        self.disconnected_lazy_times
783            .retain(|peer, disconnected_at| {
784                if now.duration_since(*disconnected_at) > retention {
785                    self.lazy_push_peers.remove(peer);
786                    false
787                } else {
788                    true
789                }
790            });
791    }
792
793    /// Moves peer into eager set with timestamp for cooldown tracking.
794    fn add_eager_at(&mut self, peer: PI, now: Instant) {
795        self.lazy_push_peers.remove(&peer);
796        self.disconnected_lazy_times.remove(&peer);
797        if self.eager_push_peers.insert(peer) {
798            self.eager_promotion_times.insert(peer, now);
799        }
800    }
801
802    /// Moves peer into lazy set and cleans up tracking state.
803    fn add_lazy(&mut self, peer: PI) {
804        self.eager_push_peers.remove(&peer);
805        self.lazy_push_peers.insert(peer);
806        self.duplicate_counts.remove(&peer);
807        self.eager_promotion_times.remove(&peer);
808    }
809
810    /// Check if peer is within prune cooldown period.
811    fn is_in_prune_cooldown(&self, peer: &PI, now: Instant) -> bool {
812        self.eager_promotion_times
813            .get(peer)
814            .map(|promotion_time| {
815                let elapsed = now.saturating_duration_since(*promotion_time);
816                elapsed < self.config.prune_cooldown
817            })
818            .unwrap_or(false)
819    }
820
821    /// Remove expired promotion time entries.
822    fn expire_promotion_times(&mut self, now: Instant) {
823        self.eager_promotion_times.retain(|_peer, promotion_time| {
824            let elapsed = now.saturating_duration_since(*promotion_time);
825            elapsed < self.config.prune_cooldown
826        });
827    }
828
829    /// Handle duplicate message receipt - decide whether to prune based on config.
830    fn maybe_prune_on_duplicate(
831        &mut self,
832        sender: PI,
833        message_id: &MessageId,
834        now: Instant,
835        io: &mut impl IO<PI>,
836    ) {
837        // Never prune for relays of our own messages
838        if self.originated_messages.contains_key(message_id) {
839            debug!("skip prune: originated message");
840            return;
841        }
842
843        // Don't prune during cooldown period
844        if self.is_in_prune_cooldown(&sender, now) {
845            debug!("skip prune: cooldown");
846            return;
847        }
848
849        // Don't prune below minimum eager peers
850        if self.eager_push_peers.len() <= self.config.min_eager_peers {
851            debug!("skip prune: min_eager_peers");
852            return;
853        }
854
855        // Check prune mode
856        match self.config.prune_mode {
857            PruneMode::Immediate => {
858                debug!("PRUNE: immediate mode");
859                self.add_lazy(sender);
860                io.push(OutEvent::SendMessage(sender, Message::Prune));
861            }
862            PruneMode::AfterDuplicates(threshold) => {
863                let count = self.duplicate_counts.entry(sender).or_insert(0);
864                *count += 1;
865                debug!("duplicate count: {}/{}", count, threshold);
866                if *count >= threshold {
867                    debug!("PRUNE: after {} duplicates", count);
868                    self.add_lazy(sender);
869                    io.push(OutEvent::SendMessage(sender, Message::Prune));
870                }
871            }
872            PruneMode::Disabled => {
873                // Don't prune
874            }
875        }
876    }
877
878    /// Immediately sends message to eager peers.
879    fn eager_push(&mut self, gossip: Gossip, sender: &PI, io: &mut impl IO<PI>) {
880        for peer in self
881            .eager_push_peers
882            .iter()
883            .filter(|peer| **peer != self.me && *peer != sender)
884        {
885            io.push(OutEvent::SendMessage(
886                *peer,
887                Message::Gossip(gossip.clone()),
888            ));
889        }
890    }
891
892    /// Queue lazy message announcements into the queue that will be sent out as batched
893    /// [`Message::IHave`] messages once the [`Timer::DispatchLazyPush`] timer is triggered.
894    fn lazy_push(&mut self, gossip: Gossip, sender: &PI, io: &mut impl IO<PI>) {
895        let Some(round) = gossip.round() else {
896            return;
897        };
898        for peer in self.lazy_push_peers.iter().filter(|x| *x != sender) {
899            self.lazy_push_queue.entry(*peer).or_default().push(IHave {
900                id: gossip.id,
901                round,
902            });
903        }
904        if !self.dispatch_timer_scheduled {
905            io.push(OutEvent::ScheduleTimer(
906                self.config.dispatch_timeout,
907                Timer::DispatchLazyPush,
908            ));
909            self.dispatch_timer_scheduled = true;
910        }
911    }

1315

1317        for i in 0..3 {
1318            let content: Bytes = format!("msg{i}").into_bytes().into();
1319            let id = MessageId::from_content(&content);
1320
1321            // First, receive from peer 3 (original)
1322            state.handle(
1323                InEvent::RecvMessage(
1324                    3,
1325                    Message::Gossip(Gossip {
1326                        id,
1327                        content: content.clone(),
1328                        scope: DeliveryScope::Swarm(Round(1)),
1329                    }),
1330                ),
1331                now,
1332                &mut io,
1333            );
1334            io.clear();
1335
1336            // Then receive duplicate from peer 2
1337            state.handle(
1338                InEvent::RecvMessage(
1339                    2,
1340                    Message::Gossip(Gossip {
1341                        id,
1342                        content: content.clone(),
1343                        scope: DeliveryScope::Swarm(Round(2)),
1344                    }),
