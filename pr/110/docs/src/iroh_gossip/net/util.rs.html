56#[derive(Debug, Serialize, Deserialize)]
57pub(crate) struct StreamHeader {
58    pub(crate) topic_id: TopicId,
59}
60
61impl StreamHeader {
62    pub(crate) async fn read(
63        stream: &mut RecvStream,
64        buffer: &mut BytesMut,
65        max_message_size: usize,
66    ) -> Result<Self, ReadError> {
67        let header: Self = read_frame(stream, buffer, max_message_size)
68            .await?
69            .ok_or_else(|| {
70                ReadError::from(io::Error::new(
71                    io::ErrorKind::UnexpectedEof,
72                    "stream ended before header",
73                ))
74            })?;
75        Ok(header)
76    }
77
78    pub(crate) async fn write(
79        self,
80        stream: &mut SendStream,
81        buffer: &mut Vec<u8>,
82        max_message_size: usize,
83    ) -> Result<(), WriteError> {
84        write_frame(stream, &self, buffer, max_message_size).await?;
85        Ok(())
86    }
87}

89pub(crate) struct RecvLoop {
90    remote_endpoint_id: EndpointId,
91    conn: Connection,
92    max_message_size: usize,
93    in_event_tx: mpsc::Sender<InEvent>,
94}
95
96impl RecvLoop {
97    pub(crate) fn new(
98        remote_endpoint_id: EndpointId,
99        conn: Connection,
100        in_event_tx: mpsc::Sender<InEvent>,
101        max_message_size: usize,
102    ) -> Self {
103        Self {
104            remote_endpoint_id,
105            conn,
106            max_message_size,
107            in_event_tx,
108        }
109    }
110
111    pub(crate) async fn run(&mut self) -> Result<(), ReadError> {
112        let mut read_futures = FuturesUnordered::new();
113        let mut conn_is_closed = false;
114        let closed = self.conn.closed();
115        tokio::pin!(closed);
116        while !conn_is_closed || !read_futures.is_empty() {
117            tokio::select! {
118                _ = &mut closed, if !conn_is_closed => {
119                    conn_is_closed = true;
120                }
121                stream = self.conn.accept_uni(), if !conn_is_closed => {
122                    let stream = match stream {
123                        Ok(stream) => stream,
124                        Err(_) => {
125                            conn_is_closed = true;
126                            continue;
127                        }
128                    };
129                    let state = RecvStreamState::new(stream, self.max_message_size).await?;
130                    debug!(topic=%state.header.topic_id.fmt_short(), "stream opened");
131                    read_futures.push(state.next());
132                }
133                Some(res) = read_futures.next(), if !read_futures.is_empty() => {
134                    let (state, msg) = match res {
135                        Ok((state, msg)) => (state, msg),
136                        Err(err) => {
137                            debug!("recv stream closed with error: {err:#}");
138                            continue;
139                        }
140                    };
141                    match msg {
142                        None => debug!(topic=%state.header.topic_id.fmt_short(), "stream closed"),
143                        Some(msg) => {
144                            if self.in_event_tx.send(InEvent::RecvMessage(self.remote_endpoint_id, msg)).await.is_err() {
145                                debug!("stop recv loop: actor closed");
146                                break;
147                            }
148                            read_futures.push(state.next());
149                        }
150                    }
151                }
152            }
153        }
154        debug!("recv loop closed");
155        Ok(())
156    }
157}

159#[derive(Debug)]
160struct RecvStreamState {
161    stream: RecvStream,
162    header: StreamHeader,
163    buffer: BytesMut,
164    max_message_size: usize,
165}
166
167impl RecvStreamState {
168    async fn new(mut stream: RecvStream, max_message_size: usize) -> Result<Self, ReadError> {
169        let mut buffer = BytesMut::new();
170        let header = StreamHeader::read(&mut stream, &mut buffer, max_message_size).await?;
171        Ok(Self {
172            buffer: BytesMut::new(),
173            max_message_size,
174            stream,
175            header,
176        })
177    }
178
179    /// Reads the next message from the stream.
180    ///
181    /// Returns `self` and the next message, or `None` if the stream ended gracefully.
182    ///
183    /// ## Cancellation safety
184    ///
185    /// This function is not cancellation-safe.
186    async fn next(mut self) -> Result<(Self, Option<ProtoMessage>), ReadError> {
187        let msg = read_frame(&mut self.stream, &mut self.buffer, self.max_message_size).await?;
188        let msg = msg.map(|msg| ProtoMessage {
189            topic: self.header.topic_id,
190            message: msg,
191        });
192        Ok((self, msg))
193    }
194}

196pub(crate) struct SendLoop {
197    conn: Connection,
198    streams: HashMap<TopicId, SendStream>,
199    buffer: Vec<u8>,
200    max_message_size: usize,
201    finishing: JoinSet<()>,
202    send_rx: mpsc::Receiver<ProtoMessage>,
203}
204
205impl SendLoop {
206    pub(crate) fn new(
207        conn: Connection,
208        send_rx: mpsc::Receiver<ProtoMessage>,
209        max_message_size: usize,
210    ) -> Self {
211        Self {
212            conn,
213            max_message_size,
214            buffer: Default::default(),
215            streams: Default::default(),
216            finishing: Default::default(),
217            send_rx,
218        }
219    }
220
221    pub(crate) async fn run(&mut self, queue: Vec<ProtoMessage>) -> Result<(), WriteError> {
222        for msg in queue {
223            self.write_message(&msg).await?;
224        }
225        let conn_clone = self.conn.clone();
226        let closed = conn_clone.closed();
227        tokio::pin!(closed);
228        loop {
229            tokio::select! {
230                biased;
231                _ = &mut closed => break,
232                Some(msg) = self.send_rx.recv() => self.write_message(&msg).await?,
233                _ = self.finishing.join_next(), if !self.finishing.is_empty() => {}
234                else => break,
235            }
236        }
237
238        // Close remaining streams.
239        for (topic_id, mut stream) in self.streams.drain() {
240            stream.finish().ok();
241            self.finishing.spawn(
242                async move {
243                    stream.stopped().await.ok();
244                    debug!(topic=%topic_id.fmt_short(), "stream closed");
245                }
246                .instrument(tracing::Span::current()),
247            );
248        }
249        if !self.finishing.is_empty() {
250            trace!(
251                "send loop closing, waiting for {} send streams to finish",
252                self.finishing.len()
253            );
254            // Wait for the remote to acknowledge all streams are finished.
255            if let Err(_elapsed) = n0_future::time::timeout(Duration::from_secs(5), async move {
256                while self.finishing.join_next().await.is_some() {}
257            })
258            .await
259            {
260                debug!("not all send streams finished within timeout, abort")
261            }
262        }
263        debug!("send loop closed");
264        Ok(())
265    }
266
267    /// Write a [`ProtoMessage`] as a length-prefixed, postcard-encoded message on its stream.
268    ///
269    /// If no stream is opened yet, this opens a new stream for the topic and writes the topic header.
270    ///
271    /// This function is not cancellation-safe.
272    pub async fn write_message(&mut self, message: &ProtoMessage) -> Result<(), WriteError> {
273        let ProtoMessage { topic, message } = message;
274        let topic_id = *topic;
275        let is_last = message.is_disconnect();
276
277        let mut entry = match self.streams.entry(topic_id) {
278            hash_map::Entry::Occupied(entry) => entry,
279            hash_map::Entry::Vacant(entry) => {
280                let mut stream = self.conn.open_uni().await?;
281                let header = StreamHeader { topic_id };
282                header
283                    .write(&mut stream, &mut self.buffer, self.max_message_size)
284                    .await?;
285                debug!(topic=%topic_id.fmt_short(), "stream opened");
286                entry.insert_entry(stream)
287            }
288        };
289        let stream = entry.get_mut();
290
291        write_frame(stream, message, &mut self.buffer, self.max_message_size).await?;
292
293        if is_last {
294            trace!(topic=%topic_id.fmt_short(), "stream closing");
295            let mut stream = entry.remove();
296            if stream.finish().is_ok() {
297                self.finishing.spawn(
298                    async move {
299                        stream.stopped().await.ok();
300                        debug!(topic=%topic_id.fmt_short(), "stream closed");
301                    }
302                    .instrument(tracing::Span::current()),
303                );
304            }
305        }
306
307        Ok(())
308    }
309}

311/// Errors related to message reading
312#[allow(missing_docs)]
313#[common_fields({
314    backtrace: Option<snafu::Backtrace>,
315})]
316#[derive(Debug, Snafu)]
317#[snafu(module)]
318#[non_exhaustive]
319pub(crate) enum ReadError {
320    /// Deserialization failed
321    #[snafu(transparent)]
322    De { source: postcard::Error },
323    /// IO error
324    #[snafu(transparent)]
325    Io { source: std::io::Error },
326    /// Message was larger than the configured maximum message size
327    #[snafu(display("message too large"))]
328    TooLarge {},
329}

331/// Read a length-prefixed frame and decode with postcard.
332pub async fn read_frame<T: DeserializeOwned>(
333    reader: &mut RecvStream,
334    buffer: &mut BytesMut,
335    max_message_size: usize,
336) -> Result<Option<T>, ReadError> {
337    match read_lp(reader, buffer, max_message_size).await? {
338        None => Ok(None),
339        Some(data) => {
340            let message = postcard::from_bytes(&data)?;
341            Ok(Some(message))
342        }
343    }
344}
345
346/// Reads a length prefixed buffer.
347///
348/// Returns the frame as raw bytes.  If the end of the stream is reached before
349/// the frame length starts, `None` is returned.
350pub async fn read_lp(
351    reader: &mut RecvStream,
352    buffer: &mut BytesMut,
353    max_message_size: usize,
354) -> Result<Option<Bytes>, ReadError> {
355    let size = match reader.read_u32().await {
356        Ok(size) => size,
357        Err(err) if err.kind() == io::ErrorKind::UnexpectedEof => return Ok(None),
358        Err(err) => return Err(err.into()),
359    };
360    let size = usize::try_from(size).map_err(|_| read_error::TooLargeSnafu.build())?;
361    if size > max_message_size {
362        return Err(read_error::TooLargeSnafu.build());
363    }
364    buffer.resize(size, 0u8);
365    reader
366        .read_exact(&mut buffer[..])
367        .await
368        .map_err(io::Error::other)?;
369    Ok(Some(buffer.split_to(size).freeze()))
370}
371
372/// Writes a length-prefixed frame.
373pub async fn write_frame<T: Serialize>(
374    stream: &mut SendStream,
375    message: &T,
376    buffer: &mut Vec<u8>,
377    max_message_size: usize,
378) -> Result<(), WriteError> {
379    let len = postcard::experimental::serialized_size(&message)?;
380    if len >= max_message_size {
381        return Err(write_error::TooLargeSnafu.build());
382    }
383    buffer.clear();
384    buffer.resize(len, 0u8);
385    let slice = postcard::to_slice(&message, buffer)?;
386    stream.write_u32(len as u32).await?;
387    stream.write_all(slice).await.map_err(io::Error::other)?;
388    Ok(())
389}

391/// A [`TimerMap`] with an async method to wait for the next timer expiration.
392#[derive(Debug)]
393pub struct Timers<T> {
394    map: TimerMap<T>,
395}
396
397impl<T> Default for Timers<T> {
398    fn default() -> Self {
399        Self {
400            map: TimerMap::default(),
401        }
402    }
403}
404
405impl<T> Timers<T> {
406    /// Creates a new timer map.
407    pub fn new() -> Self {
408        Self::default()
409    }
410
411    /// Inserts a new entry at the specified instant
412    pub fn insert(&mut self, instant: Instant, item: T) {
413        self.map.insert(instant, item);
414    }
415
416    /// Waits for the next timer to elapse.
417    pub async fn wait_next(&mut self) -> Instant {
418        match self.map.first() {
419            None => std::future::pending::<Instant>().await,
420            Some(instant) => {
421                sleep_until(*instant).await;
422                *instant
423            }
424        }
425    }
426
427    /// Pops the earliest timer that expires at or before `now`.
428    pub fn pop_before(&mut self, now: Instant) -> Option<(Instant, T)> {
429        self.map.pop_before(now)
430    }
