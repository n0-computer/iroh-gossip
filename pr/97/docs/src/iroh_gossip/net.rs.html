66/// Publish and subscribe on gossiping topics.
67///
68/// Each topic is a separate broadcast tree with separate memberships.
69/// A topic has to be joined before you can publish or subscribe on the topic.
70/// To join the swarm for a topic, you have to know the [`NodeId`] of at least one peer that also joined the topic.
71///
72/// Messages published on the swarm will be delivered to all peers that joined the swarm for that
73/// topic. You will also be relaying (gossiping) messages published by other peers.
74///
75/// With the default settings, the protocol will maintain up to 5 peer connections per topic.
76///
77/// Even though the [`Gossip`] is created from a [`Endpoint`], it does not accept connections
78/// itself. You should run an accept loop on the [`Endpoint`] yourself, check the ALPN protocol of incoming
79/// connections, and if the ALPN protocol equals [`GOSSIP_ALPN`], forward the connection to the
80/// gossip actor through [Self::handle_connection].
81///
82/// The gossip actor will, however, initiate new connections to other peers by itself.
83#[derive(Debug, Clone)]
84pub struct Gossip(Arc<Inner>);

213        let metrics = Arc::new(Metrics::default());
214        let max_message_size = config.max_message_size;
215        let me = endpoint.node_id();
216        let (api_tx, local_tx, actor) = Actor::new(endpoint, config, alpn, metrics.clone());

224            local_tx,
225            max_message_size,
226            api: GossipApi::local(api_tx),
227            metrics,
228            _actor_handle: AbortOnDropHandle::new(actor_task),

373    async fn tick(&mut self) -> bool {
374        trace!("wait for tick");
375        self.metrics.actor_tick_main.inc();
376        tokio::select! {
377            addr = self.node_addr_updates.next() => {
378                trace!("tick: node_addr_update");
379                match addr {
380                    None => {
381                        warn!("address stream returned None - endpoint has shut down");
382                        false
383                    }
384                    Some(addr) => {
385                        let data = addr.map(|addr| AddrInfo::from(addr).encode());
386                        self.our_peer_data.set(data).ok();
387                        true
388                    }
389                }
390            }
391            Some(msg) = self.local_rx.recv() => {
392                trace!("tick: local_rx {msg:?}");
393                match msg {
394                    LocalActorMessage::HandleConnection(node_id, connection) => {
395                        self.handle_remote_connection(node_id, Ok(connection), Direction::Accept).await;
396                    }
397                    LocalActorMessage::Connect(node_id, topic_id) => {
398                        self.connect(node_id, topic_id);
399                    }
400                    LocalActorMessage::SetPeerData(node_id, data) => {
401                        match AddrInfo::decode(&data) {
402                            Err(err) => warn!(remote=%node_id.fmt_short(), ?err, len=data.inner().len(), "Failed to decode peer data"),
403                            Ok(info) => {
404                                debug!(peer = ?node_id, "add known addrs: {info:?}");
405                                let node_addr = info.into_node_addr(node_id);
406                                if let Err(err) = self
407                                    .endpoint()
408                                    .add_node_addr_with_source(node_addr, DISCOVERY_PROVENANCE)
409                                {
410                                    debug!(remote=%node_id.fmt_short(), "add addr failed: {err:?}");
411                                }
412                            }
413                        }
414                    }
415                }
416                true
417            }
418            (node_id, res) = self.dialer.next_conn() => {
419                trace!(remote=%node_id.fmt_short(), ok=res.is_ok(), "tick: dialed");
420                self.handle_remote_connection(node_id, res, Direction::Dial).await;
421                true
422            }
423            Some((node_id, res)) = self.accepting.next(), if !self.accepting.is_empty() => {
424                trace!(remote=%node_id.fmt_short(), res=?res.as_ref().map(|_| ()), "tick: accepting");
425                match res {
426                    Ok(request) => self.handle_remote_message(node_id, request).await,
427                    Err(reason) => {
428                        debug!(remote=%node_id.fmt_short(), ?reason, "accept loop for remote closed");
429                    }
430                }
431                true
432            }
433            msg = self.api_rx.recv() => {
434                trace!(some=msg.is_some(), "tick: api_rx");
435                match msg {
436                    Some(msg) => {
437                        self.handle_api_message(msg).await;
438                        true
439                    }
440                    None => {
441                        trace!("all api senders dropped, stop actor");
442                        false
443                    }
444                }
445            }
446            Some(res) = self.close_connections.join_next(), if !self.close_connections.is_empty() => {
447                let (node_id, connection) = res.expect("connection task panicked");
448                trace!(remote=%node_id.fmt_short(), "tick: connection closed");
449                if let Some(state) = self.remotes.get(&node_id) {
450                    if state.same_connection(&connection) {
451                        self.remotes.remove(&node_id);
452                    }
453                }
454                true
455            }
456            Some(actor) = self.topic_tasks.join_next(), if !self.topic_tasks.is_empty() => {
457                let actor = actor.expect("topic actor task panicked");
458                trace!(topic=%actor.topic_id.fmt_short(), "tick: topic actor finished");
459                self.topics.remove(&actor.topic_id);
460                true
461            }
462            else => unreachable!("reached else arm, but all fallible cases should be handled"),
463        }
464    }

470    fn drain_pending_dials(
471        &mut self,
472        remote: &NodeId,
473    ) -> impl Iterator<Item = (TopicId, &TopicHandle)> {
474        self.topic_pending_remotes
475            .remove(remote)
476            .into_iter()
477            .flatten()
478            .flat_map(|topic_id| self.topics.get(&topic_id).map(|handle| (topic_id, handle)))
479    }

481    fn connect(&mut self, remote: NodeId, topic_id: TopicId) {
482        let Some(handle) = self.topics.get(&topic_id) else {
483            return;
484        };
485        if let Some(state) = self.remotes.get(&remote) {
486            let tx = handle.tx.clone();
487            let state = state.clone();
488            task::spawn(async move {
489                let msg = state.open_topic(topic_id).await;
490                tx.send(msg).await.ok();
491            });
492        } else {
493            self.dialer.queue_dial(remote, self.alpn.clone());
494            self.topic_pending_remotes
495                .entry(remote)
496                .or_default()
497                .insert(topic_id);
498        }
499    }

501    #[instrument("connection", skip_all, fields(remote=%remote.fmt_short()))]
502    async fn handle_remote_connection(
503        &mut self,
504        remote: NodeId,
505        res: Result<Connection, ConnectError>,
506        direction: Direction,
507    ) {
508        match (res.as_ref(), direction) {
509            (Ok(_), Direction::Dial) => inc(&self.metrics.peers_dialed_success),
510            (Err(_), Direction::Dial) => inc(&self.metrics.peers_dialed_failure),
511            (Ok(_), Direction::Accept) => inc(&self.metrics.peers_accepted),
512            (Err(_), Direction::Accept) => {}
513        }
514        let connection = match res {
515            Err(err) => {
516                debug!(?err, "Connection failed");
517                for (_, handle) in self.drain_pending_dials(&remote) {
518                    handle
519                        .send(ActorToTopic::ConnectionFailed(remote))
520                        .await
521                        .ok();
522                }
523                return;
524            }
525            Ok(connection) => connection,
526        };
527
528        let state = RemoteState::new(remote, connection.clone(), direction);
529
530        // Open requests for pending topics.
531        for (topic_id, handle) in self.drain_pending_dials(&remote) {
532            let tx = handle.tx.clone();
533            let state = state.clone();
534            task::spawn(
535                async move {
536                    let msg = state.open_topic(topic_id).await;
537                    tx.send(msg).await.ok();
538                }
539                .instrument(tracing::Span::current()),
540            );
541        }
542
543        // Read incoming requests.
544        let counter = state.counter.clone();
545        self.accepting.push(Box::pin(
546            accept_stream::<net_proto::Request>(connection.clone())
547                .map(move |req| (remote, req.map(|r| counter.guard(r)))),
548        ));
549
550        // Close on idle (if dialed) or await close (if accepted).
551        let counter = state.counter.clone();
552        let fut = async move {
553            match direction {
554                Direction::Dial => {
555                    counter.idle_for(Duration::from_millis(500)).await;
556                    info!("close connection (from dial): unused");
557                    connection.close(1u32.into(), b"idle");
558                }
559                Direction::Accept => {
560                    let reason = connection.closed().await;
561                    info!(?reason, "connection closed (from accept)")
562                }
563            }
564            (remote, connection)
565        };
566        self.close_connections
567            .spawn(fut.instrument(error_span!("conn", remote=%remote.fmt_short())));
568
569        self.remotes.insert(remote, state);
570    }
571
572    #[instrument("request", skip_all, fields(remote=%remote.fmt_short()))]
573    async fn handle_remote_message(&mut self, remote: NodeId, request: Guarded<GossipMessage>) {
574        let (request, guard) = request.split();
575        let (topic_id, request) = match request {
576            GossipMessage::Join(req) => (req.inner.topic_id, req),
577        };
578        if let Some(topic) = self.topics.get(&topic_id) {
579            if let Err(_err) = topic
580                .send(ActorToTopic::Connected {
581                    remote,
582                    tx: Guarded::new(request.tx, guard.clone()),
583                    rx: Guarded::new(request.rx, guard.clone()),
584                })
585                .await
586            {
587                warn!(topic=%topic_id.fmt_short(), "Topic actor dead");
588            }
589        } else {
590            debug!(topic=%topic_id.fmt_short(), "ignore request: unknown topic");
591        }
592    }
593
594    async fn handle_api_message(&mut self, msg: api::RpcMessage) {
595        let (topic_id, msg) = match msg {
596            api::RpcMessage::Join(msg) => (msg.inner.topic_id, msg),
597        };
598        let topic = self.topics.entry(topic_id).or_insert_with(|| {
599            let (handle, actor) = TopicHandle::new(
600                self.me,
601                topic_id,
602                self.config.clone(),
603                self.local_tx.clone(),
604                self.our_peer_data.watch(),
605                self.metrics.clone(),
606            );
607            self.topic_tasks.spawn(
608                actor
609                    .run()
610                    .instrument(error_span!("topic", topic=%topic_id.fmt_short())),
611            );
612            handle
613        });
614        if topic.send(ActorToTopic::Api(msg)).await.is_err() {
615            warn!(topic=%topic_id.fmt_short(), "Topic actor dead");
616        }
617    }

620#[derive(Clone)]
621struct RemoteState {
622    node_id: NodeId,
623    conn_id: usize,
624    client: irpc::Client<net_proto::Request>,
625    #[allow(dead_code)]
626    direction: Direction,
627    counter: ConnectionCounter,
628}
629
630impl RemoteState {
631    fn new(node_id: NodeId, connection: Connection, direction: Direction) -> Self {
632        let conn_id = connection.stable_id();
633        let irpc_conn = IrohRemoteConnection::new(connection);
634        let client = irpc::Client::boxed(irpc_conn);
635        let counter = ConnectionCounter::new();
636        RemoteState {
637            client,
638            direction,
639            conn_id,
640            counter,
641            node_id,
642        }
643    }
644
645    fn same_connection(&self, conn: &Connection) -> bool {
646        self.conn_id == conn.stable_id()
647    }
648
649    async fn open_topic(&self, topic_id: TopicId) -> ActorToTopic {
650        let guard = self.counter.get_one();
651        let req = net_proto::JoinRequest { topic_id };
652        match self.client.bidi_streaming(req.clone(), 64, 64).await {
653            Ok((tx, rx)) => ActorToTopic::Connected {
654                remote: self.node_id,
655                tx: Guarded::new(tx, guard.clone()),
656                rx: Guarded::new(rx, guard),
657            },
658            Err(err) => {
659                warn!(?topic_id, ?err, "failed to open stream with remote");
660                ActorToTopic::ConnectionFailed(self.node_id)
661            }
662        }
663    }
664}

734    state: State,

736    timers: Timers<Timer>,
737    neighbors: BTreeSet<NodeId>,
738    peer_data: Direct<Option<PeerData>>,
739    out_events: VecDeque<OutEvent>,
740    api_receivers: MergeUnbounded<ApiRecvStream>,
741    remote_senders: HashMap<NodeId, MaybeSender>,
742    remote_receivers: MergeUnbounded<RemoteRecvStream>,
743    forward_event_tx: broadcast::Sender<ProtoEvent>,
744    forward_event_tasks: JoinSet<()>,
745    #[cfg(test)]
746    joined: Arc<AtomicBool>,
747    init: bool,
748    drop_peers_queue: HashSet<NodeId>,
749    metrics: Arc<Metrics>,

752impl TopicActor {
753    pub async fn run(mut self) -> Self {
754        self.metrics.topics_joined.inc();
755        let peer_data = self.peer_data.clone().stream();
756        tokio::pin!(peer_data);
757        loop {
758            tokio::select! {
759                Some(msg) = self.actor_rx.recv() => {
760                    trace!("tick: actor_rx {msg}");
761                    self.handle_actor_message(msg).await;
762                },
763                Some(cmd) = self.api_receivers.next(), if !self.api_receivers.is_empty() => {
764                    self.handle_api_command(cmd).await;
765                }
766                Some((remote, message)) = self.remote_receivers.next(), if !self.remote_receivers.is_empty() => {
767                    trace!(remote=%remote.fmt_short(), msg=?message, "tick: remote_rx");
768                    self.handle_remote_message(remote, message).await;
769                }
770                Some(data) = peer_data.next() => {
771                    if let Some(data) = data {
772                        self.handle_in_event(InEvent::UpdatePeerData(data)).await;
773                    }
774                }
775                _ = self.timers.wait_next() => {
776                    let now = Instant::now();
777                    while let Some((_instant, timer)) = self.timers.pop_before(now) {
778                        self.handle_in_event(InEvent::TimerExpired(timer)).await;
779                    }
780                }
781                _ = self.forward_event_tasks.join_next(), if !self.forward_event_tasks.is_empty() => {}
782                else => break,
783            }
784
785            if !self.drop_peers_queue.is_empty() {
786                let now = Instant::now();
787                for peer in self.drop_peers_queue.drain() {
788                    self.out_events
789                        .extend(self.state.handle(InEvent::PeerDisconnected(peer), now));
790                }
791                self.process_out_events(now).await;
792            }
793
794            if self.to_actor_tx.is_closed() {
795                warn!("Channel to main actor closed: abort topic loop");
796                break;
797            }
798            if self.init && self.api_receivers.is_empty() && self.forward_event_tasks.is_empty() {
799                debug!("Closing topic: All API subscribers dropped");
800                break;
801            }
802        }
803        self.metrics.topics_quit.inc();
804        self
805    }
806
807    async fn handle_actor_message(&mut self, msg: ActorToTopic) {
808        match msg {
809            ActorToTopic::Connected { remote, rx, tx } => {
810                self.remote_receivers
811                    .push(Box::pin(into_stream(rx).map(move |msg| (remote, msg))));
812                let sender = self.remote_senders.entry(remote).or_default();
813                if let Err(err) = sender.init(tx).await {
814                    warn!("Remote failed while pushing queued messages: {err:?}");
815                }
816            }
817            ActorToTopic::Api(req) => {
818                self.init = true;
819                let WithChannels { inner, tx, rx, .. } = req;
820                let initial_neighbors = self.neighbors.clone().into_iter();
821                self.forward_event_tasks.spawn(
822                    forward_events(tx, self.forward_event_tx.subscribe(), initial_neighbors)
823                        .instrument(tracing::Span::current()),
824                );
825                self.api_receivers.push(Box::pin(into_stream2(rx)));
826                self.handle_in_event(InEvent::Command(Command::Join(
827                    inner.bootstrap.into_iter().collect(),
828                )))
829                .await;
830            }
831            ActorToTopic::ConnectionFailed(node_id) => {
832                self.handle_in_event(InEvent::PeerDisconnected(node_id))
833                    .await
834            }
835        }
836    }
837
838    async fn handle_remote_message(
839        &mut self,
840        remote: NodeId,
841        message: Result<Option<ProtoMessage>, RecvError>,
842    ) {
843        let event = match message {
844            Ok(Some(message)) => InEvent::RecvMessage(remote, message),
845            Ok(None) => {
846                debug!(remote=%remote.fmt_short(), "Recv stream from remote closed");
847                InEvent::PeerDisconnected(remote)
848            }
849            Err(err) => {
850                warn!(remote=%remote.fmt_short(), ?err, "Recv stream from remote failed");
851                InEvent::PeerDisconnected(remote)
852            }
853        };
854        self.handle_in_event(event).await;
855    }
856
857    async fn handle_api_command(&mut self, command: Result<api::Command, channel::RecvError>) {
858        let Ok(command) = command else {
859            return;
860        };
861        trace!("tick: api command {command}");
862        self.handle_in_event(InEvent::Command(command.into())).await;
863    }
864
865    async fn handle_in_event(&mut self, event: InEvent) {
866        trace!("tick: in event {event:?}");
867        let now = Instant::now();
868        self.metrics.track_in_event(&event);
869        self.out_events.extend(self.state.handle(event, now));
870        self.process_out_events(now).await;
871    }
872
873    async fn process_out_events(&mut self, now: Instant) {
874        while let Some(event) = self.out_events.pop_front() {
875            trace!("tick: out event {event:?}");
876            self.metrics.track_out_event(&event);
877            match event {
878                OutEvent::SendMessage(node_id, message) => {
879                    self.send(node_id, message).await;
880                }
881                OutEvent::EmitEvent(event) => {
882                    self.handle_event(event);
883                }
884                OutEvent::ScheduleTimer(delay, timer) => {
885                    self.timers.insert(now + delay, timer);
886                }
887                OutEvent::DisconnectPeer(node_id) => {
888                    self.remote_senders.remove(&node_id);
889                }
890                OutEvent::PeerData(node_id, peer_data) => {
891                    self.to_actor_tx
892                        .send(LocalActorMessage::SetPeerData(node_id, peer_data))
893                        .await
894                        .ok();
895                }
896            }
897        }
898    }
899
900    #[instrument(skip_all, fields(remote=%remote.fmt_short()))]
901    async fn send(&mut self, remote: NodeId, message: ProtoMessage) {
902        let sender = match self.remote_senders.entry(remote) {
903            hash_map::Entry::Occupied(entry) => entry.into_mut(),
904            hash_map::Entry::Vacant(entry) => {
905                debug!("requesting new connection");
906                self.to_actor_tx
907                    .send(LocalActorMessage::Connect(remote, self.topic_id))
908                    .await
909                    .ok();
910                entry.insert(Default::default())
911            }
912        };
913        if let Err(err) = sender.send(message).await {
914            warn!(?err, remote=%remote.fmt_short(), "failed to send message");
915            self.drop_peers_queue.insert(remote);
916        }
917    }
918
919    fn handle_event(&mut self, event: ProtoEvent) {
920        match &event {
921            ProtoEvent::NeighborUp(n) => {
922                #[cfg(test)]
923                self.joined
924                    .store(true, std::sync::atomic::Ordering::Relaxed);
925                self.neighbors.insert(*n);
926            }
927            ProtoEvent::NeighborDown(n) => {
928                self.neighbors.remove(n);
929            }
930            ProtoEvent::Received(_) => {}
931        }
932        self.forward_event_tx.send(event).ok();
933    }
934}
935
936async fn forward_events(
937    tx: channel::mpsc::Sender<api::Event>,
938    mut sub: broadcast::Receiver<ProtoEvent>,
939    initial_neighbors: impl Iterator<Item = NodeId>,
940) {
941    for neighbor in initial_neighbors {
942        if let Err(_err) = tx.send(api::Event::NeighborUp(neighbor)).await {
943            break;
944        }
945    }
946    loop {
947        let event = tokio::select! {
948            biased;
949            event = sub.recv() => event,
950            _ = tx.closed() => break
951        };
952        let event: api::Event = match event {
953            Ok(event) => event.into(),
954            Err(broadcast::error::RecvError::Lagged(_)) => api::Event::Lagged,
955            Err(broadcast::error::RecvError::Closed) => break,
956        };
957        if let Err(_err) = tx.send(event).await {
958            break;
959        }
960    }
961}

963#[derive(Debug)]
964enum MaybeSender {
965    Active(Guarded<channel::mpsc::Sender<ProtoMessage>>),
966    Pending(Vec<ProtoMessage>),
967}
968
969impl MaybeSender {
970    async fn send(&mut self, message: ProtoMessage) -> Result<(), channel::SendError> {
971        match self {
972            Self::Active(sender) => sender.send(message).await,
973            Self::Pending(messages) => {
974                messages.push(message);
975                Ok(())
976            }
977        }
978    }
979
980    async fn init(
981        &mut self,
982        sender: Guarded<channel::mpsc::Sender<ProtoMessage>>,
983    ) -> Result<(), channel::SendError> {
984        debug!("Initializing new sender");
985        *self = match self {
986            Self::Active(_old) => {
987                debug!("Dropping old sender");
988                Self::Active(sender)
989            }
990            Self::Pending(queue) => {
991                debug!("Sending {} queued messages", queue.len());
992                for msg in queue.drain(..) {
993                    sender.send(msg).await?;
994                }
995                Self::Active(sender)
996            }
997        };
998        Ok(())
999    }
1000}
1001
1002impl Default for MaybeSender {
1003    fn default() -> Self {
1004        Self::Pending(Vec::new())
1005    }

1008// TODO: Upstream to irpc: This differs from Receiver::into_stream: it returns
1009// None after the first error, whereas upstream would loop on the error
1010fn into_stream<T: irpc::RpcMessage>(
1011    receiver: impl DerefMut<Target = channel::mpsc::Receiver<T>> + Send + Sync + 'static,
1012) -> impl Stream<Item = Result<Option<T>, RecvError>> + Send + Sync + 'static {
1013    n0_future::stream::unfold(Some(receiver), |recv| async move {
1014        let mut recv = recv?;
1015        let res = recv.recv().await;
1016        match res {
1017            Err(err) => Some((Err(err), None)),
1018            Ok(Some(res)) => Some((Ok(Some(res)), Some(recv))),
1019            Ok(None) => Some((Ok(None), None)),
1020        }
1021    })
1022}
1023
1024fn into_stream2<T: irpc::RpcMessage>(
1025    receiver: channel::mpsc::Receiver<T>,
1026) -> impl Stream<Item = Result<T, RecvError>> + Send + Sync + 'static {
1027    n0_future::stream::unfold(Some(receiver), |recv| async move {
1028        let mut recv = recv?;
1029        match recv.recv().await {
1030            Err(err) => Some((Err(err), None)),
1031            Ok(Some(res)) => Some((Ok(res), Some(recv))),
1032            Ok(None) => None,
1033        }
1034    })
1035}

1123        endpoint: Endpoint,
1124        gossip: Gossip,
1125        cancel: CancellationToken,
1126    ) -> Result<()> {
1127        loop {
1128            tokio::select! {
1129                biased;
1130                _ = cancel.cancelled() => break,
1131                incoming = endpoint.accept() => match incoming {
1132                    None => break,
1133                    Some(incoming) => {
1134                        let connecting = match incoming.accept() {
1135                            Ok(connecting) => connecting,
1136                            Err(err) => {
1137                                warn!("incoming connection failed: {err:#}");

1144                        let remote_node_id = connection.remote_node_id()?;
1145                        gossip.handle_connection(remote_node_id, connection).await?
1146                    }
1147                }
1148            }
1149        }
1150        Ok(())

1364        }

1370                               // internal_rx connection request (from topic actor);
1371                               // dialer connected;

1378        tx.send(()).await.e()?;

1381        assert!(state.joined());

1384        tx.send(()).await.e()?;

1492    #[tokio::test]
1493    #[traced_test]
1494    async fn can_die_and_reconnect() -> Result {
1495        /// Runs a future in a separate runtime on a separate thread, cancelling everything
1496        /// abruptly once `cancel` is invoked.
1497        fn run_in_thread<T: Send + 'static>(
1498            cancel: CancellationToken,
1499            fut: impl std::future::Future<Output = T> + Send + 'static,
1500        ) -> std::thread::JoinHandle<Option<T>> {
1501            std::thread::spawn(move || {
1502                let rt = tokio::runtime::Builder::new_current_thread()
1503                    .enable_all()
1504                    .build()
1505                    .unwrap();
1506                rt.block_on(async move { cancel.run_until_cancelled(fut).await })
1507            })
1508        }
1509
1510        /// Spawns a new endpoint and gossip instance.
1511        async fn spawn_gossip(
1512            secret_key: SecretKey,
1513            relay_map: RelayMap,
1514        ) -> Result<(Router, Gossip), BindError> {
1515            let ep = Endpoint::builder()
1516                .secret_key(secret_key)
1517                .relay_mode(RelayMode::Custom(relay_map))
1518                .insecure_skip_relay_cert_verify(true)
1519                .bind()
1520                .await?;
1521            let gossip = Gossip::builder().spawn(ep.clone());
1522            let router = Router::builder(ep).accept(ALPN, gossip.clone()).spawn();
1523            Ok((router, gossip))
1524        }
1525
1526        /// Spawns a gossip node, and broadcasts a single message, then sleep until cancelled externally.
1527        async fn broadcast_once(
1528            secret_key: SecretKey,
1529            relay_map: RelayMap,
1530            bootstrap_addr: NodeAddr,
1531            topic_id: TopicId,
1532            message: String,
1533        ) -> Result {
1534            let (router, gossip) = spawn_gossip(secret_key, relay_map).await?;
1535            info!(node_id = %router.endpoint().node_id().fmt_short(), "broadcast node spawned");
1536            let bootstrap = vec![bootstrap_addr.node_id];
1537            router.endpoint().add_node_addr(bootstrap_addr)?;
1538            let mut topic = gossip.subscribe_and_join(topic_id, bootstrap).await?;
1539            topic.broadcast(message.as_bytes().to_vec().into()).await?;
1540            std::future::pending::<()>().await;
1541            Ok(())
1542        }
1543
1544        let (relay_map, _relay_url, _guard) = iroh::test_utils::run_relay_server().await.unwrap();
1545        let mut rng = &mut rand_chacha::ChaCha12Rng::seed_from_u64(1);
1546        let topic_id = TopicId::from_bytes(rng.r#gen());
1547
1548        // spawn a gossip node, send the node's address on addr_tx,
1549        // then wait to receive `count` messages, and terminate.
1550        let (addr_tx, addr_rx) = tokio::sync::oneshot::channel();
1551        let (msgs_recv_tx, mut msgs_recv_rx) = tokio::sync::mpsc::channel(3);
1552        let recv_task = tokio::task::spawn({
1553            let relay_map = relay_map.clone();
1554            let secret_key = SecretKey::generate(&mut rng);
1555            async move {
1556                let (router, gossip) = spawn_gossip(secret_key, relay_map).await?;
1557                // wait for the relay to be set. iroh currently has issues when trying
1558                // to immediately reconnect with changed direct addresses, but when the
1559                // relay path is available it works.
1560                // See https://github.com/n0-computer/iroh/pull/3372
1561                router.endpoint().home_relay().initialized().await;
1562                let addr = router.endpoint().node_addr().initialized().await;
1563                info!(node_id = %addr.node_id.fmt_short(), "recv node spawned");
1564                addr_tx.send(addr).unwrap();
1565                let mut topic = gossip.subscribe_and_join(topic_id, vec![]).await?;
1566                while let Some(event) = topic.try_next().await.unwrap() {
1567                    if let Event::Received(message) = event {
1568                        let message = std::str::from_utf8(&message.content).e()?.to_string();
1569                        msgs_recv_tx.send(message).await.e()?;
1570                    }
1571                }
1572                Result::<_, n0_snafu::Error>::Ok(())
1573            }
1574        });
1575
1576        let node0_addr = addr_rx.await.e()?;
1577        let max_wait = Duration::from_secs(5);
1578
1579        // spawn a node, send a message, and then abruptly terminate the node ungracefully
1580        // after the message was received on our receiver node.
1581        let cancel = CancellationToken::new();
1582        let secret = SecretKey::generate(&mut rng);
1583        let join_handle_1 = run_in_thread(
1584            cancel.clone(),
1585            broadcast_once(
1586                secret.clone(),
1587                relay_map.clone(),
1588                node0_addr.clone(),
1589                topic_id,
1590                "msg1".to_string(),
1591            ),
1592        );
1593        // assert that we received the message on the receiver node.
1594        let msg = timeout(max_wait, msgs_recv_rx.recv()).await.e()?.unwrap();
1595        assert_eq!(&msg, "msg1");
1596        info!("kill broadcast node");
1597        cancel.cancel();
1598
1599        // spawns the node again with the same node id, and send another message
1600        let cancel = CancellationToken::new();
1601        let join_handle_2 = run_in_thread(
1602            cancel.clone(),
1603            broadcast_once(
1604                secret.clone(),
1605                relay_map.clone(),
1606                node0_addr.clone(),
1607                topic_id,
1608                "msg2".to_string(),
1609            ),
1610        );
1611        // assert that we received the message on the receiver node.
1612        // this means that the reconnect with the same node id worked.
1613        let msg = timeout(max_wait, msgs_recv_rx.recv()).await.e()?.unwrap();
1614        assert_eq!(&msg, "msg2");
1615        info!("kill broadcast node");
1616        cancel.cancel();
1617
1618        info!("kill recv node");
1619        recv_task.abort();
1620        assert!(join_handle_1.join().unwrap().is_none());
1621        assert!(join_handle_2.join().unwrap().is_none());
1622
1623        Ok(())
1624    }
